                            req: request
                        };

                        if (options.verifyClient.length === 2) {
                            options.verifyClient(info, (result, code, name) => {
                                if (result) {

                                    abortConnection(socket, code, name);
                                }
                            });
                        } else {
                            if (options.verifyClient(info)) {
                                this.handleUpgrade(request, socket, head, emitConnection);
                            } else {
                                abortConnection(socket, 400, 'Client verification failed');

                        this.handleUpgrade(request, socket, head, emitConnection);

                        abortConnection(socket, 400, 'URL not supported');
                    }

                if (eventName === 'upgrade') {
                    this._lastUpgradeListener = false;
                }

##### Deviations from ws
There are some important incompatibilities with `ws` though, we aim to be ~90% compatible but will never implement behavior that
 is deemed too inefficient:

* Binary data is passed zero-copy as an `ArrayBuffer`. This means you need to copy it to keep it past the callback. It also means
 you need to convert it with `Buffer.from(message)` if you expect a `Node.js Buffer`.
